\begin{frame}
    \frametitle{Основные результаты работы}
    \begin{enumerate}
        \item \textbf{Математическая модель:} глобальный план $G=(\overline{\overline{X}},\overline{\overline{Y}},\overline{\overline{Z}})$, векторная целевая функция $\mu(G)=\langle\omega,\varphi,\tau\rangle\to\min$, система из~9~ограничений
        \item \textbf{Численный метод:} генетический алгоритм с~операторами мутации, глобальной мутации и~парного отбора; событийная модель расписания
        \item \textbf{Комплекс программ:} реализация модели и~численных методов для целлюлозно-бумажного производства; подтверждён результатами вычислительных экспериментов
    \end{enumerate}
    \vspace{0.3cm}
    Результаты соответствуют специальности \textbf{1.2.2}~--- <<Математическое моделирование, численные методы и~комплексы программ>>
\end{frame}
\note{
    Перечисляются основные результаты работы по трём направлениям специальности.
}

\begin{frame}
    \frametitle{Научная новизна}
    \begin{enumerate}
        \item Математическая модель, комплексно учитывающая многомашинную структуру (БДМ~$\to$ склад~$\to$ ПРС/БРС), многокритериальность ($\omega$, $\varphi$, $\tau$) и~динамику складских запасов ($\overline{T}$, $\overline{\overline{T}}$)
        \item Численный метод на~основе генетического алгоритма с~кроссинговером в~форме парного отбора, сохраняющим допустимость решений
        \item Событийная модель формирования расписания с~учётом ограничений вместимости складов
    \end{enumerate}
\end{frame}
\note{
    Проговаривается научная новизна: комплексная модель, модификация ГА и событийное расписание.
}

\begin{frame}
    \frametitle{Практическая значимость}
    \begin{itemize}
        \item Разработанные модели и~численные методы могут использоваться при создании автоматизированных систем планирования производства
        \item Комплекс программ применим на~предприятиях целлюлозно-бумажной промышленности для снижения потерь материала и~улучшения соблюдения сроков заказов
        \item Свидетельство о~регистрации программы для ЭВМ \textnumero\,2024611544
        \item Акт о внедрении
    \end{itemize}
\end{frame}
\note{
    Проговаривается практическая значимость: применимость на~предприятиях ЦБП, зарегистрированная программа.
}
