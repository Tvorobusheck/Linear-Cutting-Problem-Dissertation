\section{Введение}

\begin{frame}

    \vspace{0.3cm}
    \small
    \begin{itemize}
        \item \textbf{Бумагоделательная машина (БДМ)} выпускает тамбуры~--- широкие рулоны-полуфабрикаты фиксированной ширины
        \item \textbf{Продольно-резательный станок (ПРС)} нарезает тамбур на~рулоны заданных форматов; ограничен по~ширине и~числу ножей
        \item \textbf{Бобино-резательный станок (БРС)} нарезает узкие форматы из~рулонов, полученных на~ПРС
        \item \textbf{Склады} хранят тамбуры между выпуском и~нарезкой; ограничены по~вместимости
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Постановка задачи}
    \textbf{Требуется составить согласованный план:}
    \begin{enumerate}
        \item какие тамбуры и~в~каком порядке выпустить на~каждой БДМ
        \item как нарезать каждый тамбур на~рулоны нужных ширин на~ПРС/БРС
        \item когда выполнять каждую операцию (расписание)
    \end{enumerate}

    \vspace{0.3cm}
    \textbf{Критерии качества плана} (противоречивы):
    \begin{itemize}
        \item минимизация отходов материала при нарезке
        \item минимизация числа раскроев (загрузка станков)
        \item минимизация перестановок ножей (переналадки)
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Основные ограничения:} порядок марок бумаги, вместимость складов, наличие тамбуров для нарезки, соблюдение сроков заказов, многослойные заказы
\end{frame}

\section{Математическая модель}

\begin{frame}
    \frametitle{Искомое решение~--- глобальный план $G$}
    Требуется найти план, объединяющий выпуск тамбуров, схемы раскроя и~расписание операций, при минимизации критериев целевой функции:
    \[
    \boxed{G = \bigl(\,\overline{\overline{X}},\;\overline{\overline{Y}},\;\overline{\overline{Z}}\,\bigr) \quad\longrightarrow\quad \mu(G)\to\min}
    \]

    \vspace{0.2cm}
    \textbf{План производства} $\overline{\overline{X}} = \{\overline{X}_1,\ldots,\overline{X}_I\}$~--- для каждой из~$I$ БДМ:
    \[
    X_{i,j} = \bigl(\,{w^X}_{i,j},\;\;{f^X}_{i,j},\;\;{p^X}_{i,j}\,\bigr)
    \]
    \begin{itemize}
        \item ${w^X}_{i,j}$~--- ширина тамбура, совпадает с~шириной БДМ: ${w^X}_{i,j}={w^m}_i$
        \item ${f^X}_{i,j} \in [1,U]$~--- номер марки бумаги
        \item ${p^X}_{i,j}$~--- плотность бумаги (г/м$^2$)
    \end{itemize}
    \vspace{0.1cm}
    Порядок тамбуров в~$\overline{X}_i$ определяет последовательность выпуска на~$i$-й БДМ
\end{frame}

\begin{frame}
    \frametitle{План нарезки $\overline{\overline{Y}}$}
    $\overline{\overline{Y}} = \{\overline{Y}_1,\ldots,\overline{Y}_J\}$~--- для каждого из~$J$ ПРС:
    \[
    Y_{i,j} = \bigl(\,{y^Y}_{i,j},\;\;{f^Y}_{i,j},\;\;{p^Y}_{i,j},\;\;{l^Y}_{i,j}\,\bigr)
    \]
    \begin{itemize}
        \item ${y^Y}_{i,j} = \{y_{i,j,1},\ldots,y_{i,j,|{y^Y}_{i,j}|}\}$~--- упорядоченный набор форматов
        \item Каждый формат $y_{i,j,k} = ({r^Y}_{i,j,k},\;{g^Y}_{i,j,k})$:
        \begin{itemize}
            \item ${r^Y}_{i,j,k}$~--- ширина формата
            \item ${g^Y}_{i,j,k} \in \{0\}\cup[1,Q]$~--- номер заказа ($0$~--- промежуточный рулон для~БРС)
        \end{itemize}
        \item ${f^Y}_{i,j}\in[1,U]$~--- марка бумаги раскроя
        \item ${p^Y}_{i,j}$~--- плотность бумаги раскроя
        \item ${l^Y}_{i,j}$~--- количество слоёв (число одновременно подаваемых тамбуров)
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Расписание~--- очередь событий $\overline{\overline{Z}}$}
    \small
    Каждое событие $\overline{Z}\in\overline{\overline{Z}}$:
    \[
    \overline{Z} = \bigl(\,h,\;\overline{h},\;e,\;d,\;z,\;\overline{T},\;\overline{\overline{T}},\;\overline{t'}\,\bigr)
    \]
    \begin{itemize}
        \item $e\in\{0,1\}$~--- тип: $e{=}0$~--- выпуск тамбура, $e{=}1$~--- нарезка на~ПРС
        \item $d$~--- номер машины (БДМ при $e{=}0$, ПРС при $e{=}1$)
        \item $z$~--- номер операции в~плане $d$-й машины
        \item $h$, $\overline{h}$~--- время начала и~окончания события
        \item $\overline{T}$~--- текущее содержимое складов (доступные тамбуры)
        \item $\overline{\overline{T}}$~--- запланированное содержимое (включая тамбуры в~производстве)
        \item $\overline{t'}$~--- тамбуры, использованные при нарезке ($\varnothing$ при $e{=}0$)
    \end{itemize}
    \vspace{0.1cm}
    Событие фиксирует состояние склада~--- связывает $\overline{\overline{X}}$ и~$\overline{\overline{Y}}$ во~времени
\end{frame}

\begin{frame}
    \frametitle{Входные параметры: заказы и~константы}
    \textbf{Множество заказов} $O = \{o_1,\ldots,o_Q\}$, где каждый заказ:
    \[
    o_i = (w_i,\;q_i,\;h_i,\;f_i,\;p_i,\;l_i,\;g_i)
    \]
    \begin{itemize}
        \item $w_i$~--- ширина рулона (определяет схему раскроя)
        \item $q_i$~--- требуемое количество рулонов
        \item $h_i$~--- срок исполнения (дата и~время)
        \item $f_i\in[1,U]$~--- номер марки бумаги
        \item $p_i$~--- плотность бумаги (г/м$^2$)
        \item $l_i$~--- количество слоёв (число тамбуров на~ПРС одновременно)
        \item $g_i\in\{0,1\}$~--- индикатор: $1$~--- заказ для~БРС (узкий формат)
    \end{itemize}
    \vspace{0.1cm}
    $Q$~--- количество заказов;\ $U$~--- количество марок бумаги

    \textbf{Константы:} $B=\{b_1,\ldots,b_U\}$~--- времена смены марок ($b_f$~--- время перехода на~марку~$f$);\quad $V$~--- кромка;\quad $H$~--- начало производства
\end{frame}

\begin{frame}
    \frametitle{Входные параметры: оборудование (БДМ и~ПРС)}
    \small
    \textbf{БДМ} $M=\{m_1,\ldots,m_I\}$, где $m_i=({w^m}_i,\,{v^m}_i,\,{g^m}_i)$:
    \begin{itemize}
        \item ${w^m}_i$~--- ширина выпускаемого тамбура (задаёт верхнюю границу раскроя)
        \item ${v^m}_i$~--- время производства одного тамбура
        \item ${g^m}_i$~--- время смены плотности на~1~г/м$^2$
    \end{itemize}

    \vspace{0.3cm}
    \textbf{ПРС} $C=\{c_1,\ldots,c_J\}$, где $c_i=({w^c}_i,\,{n^c}_i,\,{v^c}_i,\,{k^c}_i,\,{g^c}_i)$:
    \begin{itemize}
        \item ${w^c}_i$~--- максимальная ширина обрабатываемого тамбура
        \item ${n^c}_i$~--- количество ножей (макс.\ число форматов в~раскрое)
        \item ${v^c}_i$~--- время нарезки одного слоя
        \item ${k^c}_i$~--- время перестановки одного ножа
        \item ${g^c}_i\in\{0,1\}$~--- индикатор: $1$~--- это БРС
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Входные параметры: склады}
    \textbf{Склады} $T=\{t_1,\ldots,t_K\}$, где $t_i=({r^T}_i,\,{u^T}_i)$:
    \begin{itemize}
        \item ${r^T}_i$~--- максимальная допустимая ширина тамбура на~складе
        \item ${u^T}_i$~--- максимальное количество тамбуров (ёмкость)
    \end{itemize}

    \vspace{0.4cm}
    \textbf{Роль склада в~модели:}
    \begin{itemize}
        \item Буфер между выпуском ($\overline{\overline{X}}$) и~нарезкой ($\overline{\overline{Y}}$)
        \item Тамбур появляется на~складе по~окончании производства на~БДМ
        \item При нарезке тамбуры изымаются со~склада
        \item $\overline{T}$~--- текущее содержимое;\quad $\overline{\overline{T}}$~--- запланированное (включая тамбуры в~производстве)
    \end{itemize}
    \vspace{0.2cm}
    Ограничения $|\overline{t}_i|\le{u^T}_i$ и~$\overline{w^T}_{i,j}\le{r^T}_i$ проверяются на~каждом шаге расписания
\end{frame}

\begin{frame}
    \frametitle{Целевая функция}
    \[
    \mu(G) = \Bigl\langle \omega(G),\;\varphi(G),\;\tau\bigl(\overline{\overline{Y}}\bigr) \Bigr\rangle \rightarrow \min
    \]
    \vspace{0.1cm}
    \textbf{1. Суммарные отходы} (разность ширин тамбуров и~форматов):
    \[
    \omega(G)=\sum_{\overline{Z}\in\overline{\overline{Z}}:\,e=1}
    \Biggl(\underbrace{\sum_{i=1}^{l^Y_{d,z}}\overline{w^t}_i}_{\text{ширины тамбуров}} - \underbrace{\sum_{i=1}^{|{y^Y}_{d,z}|} r^Y_{d,z,i}}_{\text{ширины форматов}}\Biggr)
    \]
    \textbf{2. Максимальное число раскроев на одной ПРС:}
    \[
    \varphi(G)=\max_{\overline{Z}\in\overline{\overline{Z}}:\,e=1} z
    \]
    \textbf{3. Общее количество перестановок ножей:}
    \[
    \tau\bigl(\overline{\overline{Y}}\bigr)=\sum_{i=1}^{J}\sigma\bigl(i,\,|\overline{Y}_i|\bigr)
    \]
    Минимизация~--- \textbf{лексикографическая} (приоритет: $\omega \succ \varphi \succ \tau$)
\end{frame}

\begin{frame}
    \frametitle{Подсчёт перестановок ножей $\sigma(i,j)$}
    Первый раскрой на $i$-й ПРС:
    \[
    \sigma(i,1) = |y_{i,1}| + 1
    \]
    Последующие раскрои ($j>1$):
    \[
    \sigma(i,j) = \sigma(i,j{-}1) + |y_{i,j}| - \max_n\!\Bigl\{0\le n \le \min\bigl\{|y_{i,j-1}|,\,|y_{i,j}|\bigr\} : \forall k{\le}n\;\; r^Y_{i,j-1,k}=r^Y_{i,j,k}\Bigr\}
    \]
    \vspace{0.2cm}
    \begin{itemize}
        \item Учитывается совпадение начальных форматов текущего и~предыдущего раскроев
        \item Совпадающие позиции не~требуют перестановки ножей
        \item Порядок форматов в~раскрое влияет на~значение $\tau(\overline{\overline{Y}})$
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Многослойные заказы и~БРС}
    \textbf{Слои} ($l_i$)~--- число тамбуров, одновременно подаваемых на~ПРС:
    \begin{itemize}
        \item Для нарезки раскроя $Y_{i,j}$ со~слоями ${l^Y}_{i,j}$ требуется ${l^Y}_{i,j}$ совместимых тамбуров на~складе
        \item Время нарезки пропорционально числу слоёв:
        \[
        \overline{h}= h+{v^c}_d\cdot\mathbf{{l^Y}_{d,z}}+{k^c}_d\bigl(\sigma(d,z)-\sigma(d,z{-}1)\bigr)
        \]
    \end{itemize}

    \vspace{0.3cm}
    \textbf{БРС} ($g_i{=}1$, ${g^c}_j{=}1$)~--- бобино-резательные станки:
    \begin{itemize}
        \item Нарезают не~тамбуры с~БДМ, а~рулоны с~других ПРС
        \item Формат с~${g^Y}_{i,j,k}{=}0$~--- промежуточный рулон для~БРС
        \item Добавляет дополнительный этап в~производственную цепочку
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Основные ограничения (1/3)}
    \small
    \textbf{1. Порядок марок:}
    \[
    \forall i,j:\quad {f^X}_{i,j} \le {f^X}_{i,j+1}
    \]
    \textbf{2. Допустимость раскроя по ширине и~выполнение заказов:}
    \[
    V + \sum_{k=1}^{|y_{i,j}|}{r^Y}_{i,j,k} \le {w^c}_i
    \qquad
    \forall n\in[1,Q]:\;\sum I\bigl({g^Y}_{i,j,k}=n\bigr)=q_n
    \]
    \textbf{3. Полнота расписания:}
    \[
    |\overline{\overline{Z}}| = \sum_{i=1}^{I}|\overline{X}_i| + \sum_{i=1}^{J}|\overline{Y}_i|
    \]
    \textbf{4. Соблюдение сроков:}
    $\forall\overline{Z},\;\forall i\in[1,Q]:\;\overline{h}\le h_i$
\end{frame}

\begin{frame}
    \frametitle{Основные ограничения (2/3)}
    \small
    \textbf{5--6. Каждой операции выпуска/нарезки соответствует событие:}
    \[
    \forall X_{i,j}\;\exists\;\overline{Z}:\;e{=}0,\,d{=}i,\,z{=}j
    \qquad
    \forall Y_{i,j}\;\exists\;\overline{Z}:\;e{=}1,\,d{=}i,\,z{=}j
    \]
    \textbf{7. Наличие тамбуров на складе для нарезки (с~учётом слоёв):}
    \[
    \forall\overline{Z}(e{=}1):\;\sum_{i,j} I\!\bigl(\textstyle\sum_k r^Y_{d,z,k}+V\le\overline{w^T}_{i,j}\;\wedge\;{f^Y}_{d,z}=\overline{f^T}_{i,j}\;\wedge\;{p^Y}_{d,z}=\overline{p^T}_{i,j}\bigr)=\mathbf{{l^Y}_{d,z}}
    \]
    Число совместимых тамбуров на~складе $=$ числу слоёв ${l^Y}_{d,z}$
\end{frame}

\begin{frame}
    \frametitle{Основные ограничения (3/3)}
    \small
    \textbf{8--9. Вместимость складов} (текущее $\overline{T}$ и~планируемое $\overline{\overline{T}}$):
    \[
    \overline{w^T}_{i,j}\le{r^T}_i,\quad |\overline{t}_i|\le{u^T}_i
    \qquad
    \overline{\overline{w^T}}_{i,j}\le{r^T}_i,\quad |\overline{\overline{t}}_i|\le{u^T}_i
    \]

    \vspace{0.3cm}
    \textbf{Взаимосвязь ограничений:}
    \begin{itemize}
        \item Ограничения 1--2 определяют \textbf{допустимость планов} $\overline{\overline{X}}$ и~$\overline{\overline{Y}}$
        \item Ограничения 3--6 обеспечивают \textbf{полноту расписания} $\overline{\overline{Z}}$
        \item Ограничение 7 связывает \textbf{нарезку со~складом}:
              многослойные заказы ($l_i>1$) усиливают эту зависимость
        \item Ограничения 8--9 обеспечивают \textbf{логистическую реализуемость}
    \end{itemize}
\end{frame}

\section{Численные методы}

\begin{frame}
    \frametitle{Сложность задачи и~выбор метода}
    \textbf{Полный перебор:}
    \[
    T(Q,J) \le J^Q \quad\text{--- экспоненциальный рост}
    \]
    Пригоден только для задач малой размерности (эталон).

    \vspace{0.4cm}
    \textbf{Генетический алгоритм:}
    \begin{itemize}
        \item Особь~= глобальный план $G=(\overline{\overline{X}},\overline{\overline{Y}},\overline{\overline{Z}})$
        \item Каждая итерация не~ухудшает решение
        \item Число поколений: $\lceil\log_2 P\rceil$
        \item Гибкое управление временем/качеством через размер популяции $P$
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Связь компонент плана $\overline{\overline{X}}$, $\overline{\overline{Y}}$, $\overline{\overline{Z}}$}
    Компоненты глобального плана \textbf{взаимозависимы}:

    \vspace{0.2cm}
    \[
    \overline{\overline{Y}} \;\xrightarrow{\text{определяет потребность}}\; \overline{\overline{X}} \;\xrightarrow[\text{согласование}]{\text{склад }\overline{T}}\; \overline{\overline{Z}}
    \]

    \begin{itemize}
        \item Каждый раскрой $Y_{i,j}$ требует тамбур совместимой ширины, марки и~плотности
        \item Число тамбуров для $j$-го ПРС: $\sum_k |Y_{j,k}.\text{master\_rolls}|$
        \item План $\overline{\overline{X}}$ строится так, чтобы суммарный выпуск по~каждой ширине покрывал потребности $\overline{\overline{Y}}$
        \item Расписание $\overline{\overline{Z}}$ учитывает доступность тамбуров на~складе $\overline{T}$
    \end{itemize}

    \vspace{0.2cm}
    $\Rightarrow$ Изменение $\overline{\overline{Y}}$ при мутации требует пересчёта $\overline{\overline{X}}$ и~$\overline{\overline{Z}}$
\end{frame}

\begin{frame}
    \frametitle{Генетические операторы}
    \textbf{Мутация} (\textsc{Mutate}) плана $\overline{Y}_j$:
    \begin{enumerate}
        \item Перемешать форматы $y_{i,j,k}$ внутри каждого раскроя
        \item Перемешать порядок раскроев $Y_{i,j}$
    \end{enumerate}
    $\Rightarrow$ изменяет $\sigma(i,j)$ и~$\tau(\overline{\overline{Y}})$, не~нарушая ограничений по ширине

    \vspace{0.3cm}
    \textbf{Глобальная мутация} (\textsc{GlobalMutate}) плана $G$:
    \begin{enumerate}
        \item Для каждого ПРС $j=1,\ldots,J$: $\overline{Y}'_j=\textsc{Mutate}(\overline{Y}_j)$
        \item Если $\mu(G')<\mu(G)$, принять; иначе~--- отклонить
        \item Пересчитать $\overline{\overline{X}}$ и~$\overline{\overline{Z}}$
    \end{enumerate}

    \vspace{0.3cm}
    \textbf{Парный отбор}:
    \[
    \textsc{Crossingover}(G_1,G_2) = \begin{cases} G_1, & \mu(G_1)\le\mu(G_2)\\ G_2, & \text{иначе} \end{cases}
    \]
\end{frame}

\begin{frame}
    \frametitle{Пример мутации}
    \small
    Исходный план на~одном ПРС: $\overline{Y}_1=(\lambda_1,\lambda_2)$, где $\lambda_k$~--- раскрой (набор ширин форматов)
    \[
    \lambda_1=\{100,\,150,\,100\},\quad \lambda_2=\{120,\,120,\,80\}
    \]
    После \textsc{Mutate}~--- перестановка форматов и~раскроев:
    \[
    \lambda'_1=\{150,\,100,\,100\},\quad \lambda'_2=\{120,\,80,\,120\},\quad \overline{Y}'_1=(\lambda'_2,\,\lambda'_1)
    \]

    \vspace{0.2cm}
    \textbf{Эффект:}
    \begin{itemize}
        \item Набор заказов не~изменился $\Rightarrow$ ограничения по~ширине сохраняются
        \item Изменился порядок форматов $\Rightarrow$ может измениться $\sigma(i,j)$
        \item Изменился порядок раскроев $\Rightarrow$ переход $\lambda'_2\to\lambda'_1$ может дать другое число перестановок ножей
    \end{itemize}
    Принимается только при $\mu(G')<\mu(G)$
\end{frame}

\begin{frame}
    \frametitle{Схема генетического алгоритма}
    \textbf{Вход:} задача, размер популяции $P$

    \vspace{0.2cm}
    \begin{enumerate}
        \item Сформировать начальную популяцию $S_0=\{G_1,\ldots,G_P\}$ ($P$ случайных допустимых планов)
        \item \textbf{Пока} размер популяции $|S|>1$:
        \begin{enumerate}
            \item[$2.1$] Для каждого $G\in S$: $G\leftarrow\textsc{GlobalMutate}(G)$
            \item[$2.2$] Разбить $S$ на пары $(G_{2i-1},G_{2i})$
            \item[$2.3$] $S'\leftarrow\bigl\{\textsc{Crossingover}(G_{2i-1},G_{2i})\bigr\}$
            \item[$2.4$] $S\leftarrow S'$\quad ($|S|$ уменьшается вдвое)
        \end{enumerate}
        \item Вернуть $G^*$~--- оставшуюся особь
    \end{enumerate}
    \vspace{0.3cm}
    $G^*$ не~хуже любого $G\in S_0$; число итераций $\lceil\log_2 P\rceil$
\end{frame}

\begin{frame}
    \frametitle{Генерация плана производства $\overline{\overline{X}}$}
    По~сформированному $\overline{\overline{Y}}$ строится $\overline{\overline{X}}$:

    \vspace{0.3cm}
    \begin{enumerate}
        \item Для каждого ПРС $j$ подсчитать потребность в~тамбурах по~каждой комбинации (ширина, марка, плотность)
        \item Распределить тамбуры между БДМ с~учётом ${w^X}_{i,j}={w^m}_i$
        \item Упорядочить тамбуры на~каждой БДМ по~маркам:
        \[
        \forall i,j:\quad {f^X}_{i,j} \le {f^X}_{i,j+1}
        \]
    \end{enumerate}

    \vspace{0.3cm}
    \textbf{Результат:} для каждой БДМ~--- последовательность $\overline{X}_i=(X_{i,1},X_{i,2},\ldots)$, где каждый тамбур $X_{i,j}=({w^X}_{i,j},{f^X}_{i,j},{p^X}_{i,j})$
\end{frame}

\begin{frame}
    \frametitle{Генерация плана раскроя $\overline{\overline{Y}}$}
    \small
    \textbf{Шаги формирования случайного плана:}
    \begin{enumerate}
        \item Инициализировать пустые планы $\overline{Y}_1,\ldots,\overline{Y}_J$
        \item Выделить \textbf{малые заказы} ($g_i{=}1$), перемешать по~маркам
        \item Для каждого рулона малого заказа:
            \begin{itemize}
                \item Выбрать случайный БРС ($g^c_j{=}1$)
                \item Если $V + \sum_k r^Y_{j,\text{last},k} + w_i \le {w^c}_j$~--- добавить в~текущий раскрой
                \item Иначе~--- создать новый раскрой $Y_{j,\text{new}}$
            \end{itemize}
        \item Аналогично распределить \textbf{обычные заказы} ($g_i{=}0$) по~ПРС
        \item Проверить: $\forall n\in[1,Q]:\;\sum I(g^Y_{i,j,k}{=}n)=q_n$
    \end{enumerate}

    \vspace{0.2cm}
    Каждый сгенерированный план удовлетворяет ограничениям по~ширине и~ножам
\end{frame}

\begin{frame}
    \frametitle{Генерация расписания (очередь событий)}
    Каждое событие $\overline{Z}=(h,\overline{h},e,d,z,\overline{T},\overline{\overline{T}},\overline{t'})$

    \vspace{0.2cm}
    \textbf{Время окончания выпуска тамбура} ($e{=}0$):
    \[
    \overline{h}= h+{v^m}_d+{g^m}_d({p^X}_{d,z}-{p^X}_{d,z-1})+b_f\cdot I(f\ne{f^X}_{d,z-1})
    \]
    \textbf{Время окончания нарезки} ($e{=}1$):
    \[
    \overline{h}= h+{v^c}_d\cdot{l^Y}_{d,z}+{k^c}_d\bigl(\sigma(d,z)-\sigma(d,z{-}1)\bigr)
    \]
    \vspace{0.2cm}
    \begin{itemize}
        \item Склад $\overline{T}$ обновляется при каждом событии
        \item Нарезка возможна только при наличии тамбуров на складе
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Динамика склада при формировании расписания}
    \small
    При каждом событии $\overline{Z}$ склад обновляется:

    \vspace{0.2cm}
    \textbf{Выпуск тамбура} ($e{=}0$): тамбур попадает в~$\overline{\overline{T}}$ (запланированное содержимое); станет доступен в~$\overline{T}$ (текущее содержимое), когда $h_{\text{след}} \ge \overline{h}_{\text{тек}}$

    \vspace{0.2cm}
    \textbf{Нарезка} ($e{=}1$): из~$\overline{T}$ изымаются ${l^Y}_{d,z}$ тамбуров, совместимых по ширине ($\overline{w^T}_{i,j}$), марке ($\overline{f^T}_{i,j}$) и~плотности ($\overline{p^T}_{i,j}$):
    \[
    \sum_k r^Y_{d,z,k}+V\le\overline{w^T}_{i,j}\;\wedge\;{f^Y}_{d,z}=\overline{f^T}_{i,j}\;\wedge\;{p^Y}_{d,z}=\overline{p^T}_{i,j}
    \]

    \vspace{0.2cm}
    \textbf{Контроль вместимости:}
    \[
    |\overline{t}_i|\le{u^T}_i,\quad |\overline{\overline{t}}_i|\le{u^T}_i \quad\text{--- на каждом шаге}
    \]

    $\Rightarrow$ расписание строится последовательно с~проверкой доступности ресурсов
\end{frame}
